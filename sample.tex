\begin{frame}[fragile]{Issues}
You may have issues with font. Go to file \verb|beamerinnerthemesimple.sty| and change font to the one that you like.
\begin{verbatim}
\setsansfont{Futura Book}
\setmonofont{Monaco}
\end{verbatim}
You may comment 
\begin{verbatim}
\newfontfamily\futurabold{Futura-Bold}
\end{verbatim}
\end{frame}
